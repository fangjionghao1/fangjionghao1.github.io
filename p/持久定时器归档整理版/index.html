// AddTimer  æ·»åŠ è®¡æ—¶å™¨ å¤„ç†æ•°æ®åº“æ•°æ®å’Œå †æ•°æ®
func AddTimer(t ...TimerStore) error {

		timerStore := t[idx] //  é˜²æ­¢ä¸‹é¢ä½¿ç”¨å®šæ—¶å™¨ä¸­çš„æŒ‡é’ˆ
		duration := time.Duration(timerStore.Interval)
		if duration == 0 && timerStore.Repeat {

		}

		}
		switch true {
		case t[idx].Id != 0: //  è¡¨ç¤ºåœ¨æ•°æ®åº“ä¹‹ä¸­
			if timerStore.FireTime.Before(time.Now()) { //  å¤„ç†æ—¶é—´é—®é¢˜
				if duration != 0 {
					timerStore.FireTime.Time = time.Now().Add(timerStore.FireTime.Sub(time.Now()) % duration).Add(duration) //  æ›´æ–°æ•°æ®åº“
				} else {
					timerStore.FireTime.Time = time.Now()
				}

					return err
				}
			} //  ä¸æ—©äºå½“å‰ä¸åšç›¸åº”å¤„ç†
		default: //  è¡¨ç¤ºä¸åœ¨æ•°æ®åº“ä¹‹ä¸­

	for _, v := range th.timers {

			ret = append(ret, v)
		}
	}
	return ret
}

// NewTimerHeap æ–°å»ºä¸€ä¸ªå°é¡¶å †å®šæ—¶å™¨
func NewTimerHeap() *TimerHeap {
	var timerHeap TimerHeap
	heap.Init(&timerHeap)
	timerHeap.nextAddSeq = 1
	timerHeap.invokeMode = defaultMode
	return &timerHeap
}

// NewTimerHeapWithGoroutine æ–°å»ºä¸€ä¸ªå°é¡¶å †å®šæ—¶å™¨, æ¯ä¸ªcallbackç‹¬ç«‹ä½¿ç”¨åç¨‹è¿è¡Œ
func NewTimerHeapWithGoroutine() *TimerHeap {
	var timerHeap TimerHeap
	heap.Init(&timerHeap)
	timerHeap.nextAddSeq = 1
	timerHeap.invokeMode = goroutineMode
	return &timerHeap

}
func (th *TimerHeap) AddTimers(timer Timer, callbackFunc CallbackFunc) *Timer {
	th.lock.Lock()
	if timer.Repeat {
		if timer.Interval < MinTimerInterval {
			timer.Interval = MinTimerInterval
		}
	}
	if timer.FireTime.IsZero() {
		timer.FireTime = time.Now().Add(timer.Interval)
	}
	timer.callback = callbackFunc
	timer.addSeq = th.nextAddSeq
	th.nextAddSeq += 1
	heap.Push(th, &timer)
	th.lock.Unlock()
	return &timer
}

// Tick å¯¹å®šæ—¶å™¨æ¨¡å—è¿›è¡Œä¸€æ¬¡Tick
func (th *TimerHeap) Tick() {
	now := time.Now()
	th.lock.Lock()
	defer th.lock.Unlock()
	for {
		// æ²¡æœ‰ä»»ä½•å®šæ—¶å™¨ï¼Œè¿”å›
		if th.Len() <= 0 {
			break
		}

		//å †é¡¶,æ²¡æœ‰åˆ°æ—¶é—´çš„å®šæ—¶å™¨ï¼Œè¿”å›
		if th.timers[0].FireTime.After(now) {
			break
		}
		// ä»heapä¸­æ‹¿å‡ºå †é¡¶
		t := heap.Pop(th).(*Timer)

		// å¿½ç•¥å·²ç»å–æ¶ˆçš„å®šæ—¶å™¨
		if t.cancelled {
			continue
		}
		// éå‘¨æœŸä»»åŠ¡ï¼Œåˆ™ä¸‹ä¸€æ¬¡è¿­ä»£ä¸­Pop()
		if !t.Repeat {
			t.cancelled = true
		}

		// åŒæ­¥åç¨‹æ‰§è¡Œå›è°ƒ
		if th.invokeMode == defaultMode {
			runCallback(t.ID, t.callback)
		}

		// å¼‚æ­¥åç¨‹æ‰§è¡Œå›è°ƒ
		if th.invokeMode == goroutineMode {
			go func() {
				runCallback(t.ID, t.callback)

			}()
		}
		// å¦‚æœæ˜¯å¾ªç¯timerå°±æŠŠTimeré‡æ–°æ”¾å›heapä¸­
		if t.Repeat {
			t.FireTime = t.FireTime.Add(t.Interval)
			if !t.FireTime.After(now) {
				t.FireTime = now.Add(t.Interval)
			}
			t.addSeq = th.nextAddSeq
			th.nextAddSeq += 1
			heap.Push(th, t)
		}
	}
}

	//}()
	go th.selfTickRoutine(tickInterval)

func (th *TimerHeap) Stop() {
	th.stop = true
}

func (th *TimerHeap) selfTickRoutine(tickInterval time.Duration) {
	if tickInterval < MinTimerInterval {
		tickInterval = MinTimerInterval
	}
	for {
		if th.stop {
			return
		}
		time.Sleep(tickInterval)
		th.Tick()
	}
}

// è¿è¡Œå®šæ—¶å™¨çš„å›è°ƒå‡½æ•°ï¼Œå¹¶æ•è·panicï¼Œå°†panicè½¬åŒ–ä¸ºé”™è¯¯è¾“å‡º
func runCallback(id uint64, callback CallbackFunc) {
	defer func() {
		err := recover()
		if err != nil {
			logger.Log.Errorf("Callback %v panic: %v\n", id, err)
			var buf [4096]byte
			n := runtime.Stack(buf[:], false)
			logger.Log.Errorf("Callback exits from panic: %s\n", string(buf[:n]))
		}
	}()
	callback()
}
